struct VSIn { float4 Position:POSITION; float2 TexCoords:TEXCOORD0; float4 Color:COLOR0; };
struct VSOut { float4 Position:POSITION; float2 TexCoords:TEXCOORD0; float4 Color:COLOR0; };

VSOut VS(VSIn input)
{
    VSOut o;
    o.Position = mul(input.Position, transformMatrix);
    o.TexCoords = input.TexCoords;
    o.Color = input.Color;
    return o;
}

float4 PS(VSOut input) : COLOR0
{
    float2 uv = input.TexCoords;
    float2 scroll = float2(uv.x * repeats + time, uv.y);
    float n = tex2D(samplerNoise, scroll * noiseScale + float2(0, time * noiseSpeed)).r;
    float warp = (n - 0.5f) * distort;
    uv.y = saturate(uv.y + warp);

    float d = abs(uv.y - 0.5f);
    float core = smoothstep(coreWidth, 0.0f, d);
    float edge = smoothstep(edgeWidth, coreWidth, d);

    float mask = tex2D(samplerTex, float2(uv.x * repeats + time, uv.y)).r;
    float flick = lerp(1.0f - flicker, 1.0f + flicker, n);

    float3 col = edgeColor * edge + coreColor * core;
    float alpha = saturate(core + edge) * mask * input.Color.a;

    return float4(col * intensity * input.Color.rgb * flick, alpha * flick);
}

Impact/Muzzle Shader (radial burst)
float time;
float intensity;
float radius;
float thickness;
float coreRadius;
float spikeThreshold;
float spikeStrength;
float noiseScale;
float noiseSpeed;
float3 coreColor;
float3 edgeColor;

texture noiseTexture;
sampler2D samplerNoise = sampler_state
{
    texture = <noiseTexture>;
    magfilter = LINEAR; minfilter = LINEAR; mipfilter = LINEAR;
    AddressU = wrap; AddressV = wrap;
};

float4 PS(float2 uv : TEXCOORD0) : COLOR0
{
    float2 p = uv - 0.5f;
    float r = length(p) * 2.0f;

    float core = smoothstep(coreRadius, 0.0f, r);
    float ring = smoothstep(radius + thickness, radius, r);

    float n = tex2D(samplerNoise, p * noiseScale + time * noiseSpeed).r;
    float spikes = smoothstep(spikeThreshold, 1.0f, n) * ring * spikeStrength;

    float3 col = coreColor * core + edgeColor * (ring + spikes);
    float alpha = saturate(core + ring + spikes) * intensity;

    return float4(col, alpha);
}

void DrawBeam(Vector2 start, Vector2 end, BeamPreset p, Texture2D ramp, Texture2D noise)
{
    int count = p.Points;
    if (count < 2) count = 2;

    Vector2[] positions = new Vector2[count];
    for (int i = 0; i < count; i++)
        positions[i] = Vector2.Lerp(end, start, i / (float)(count - 1));

    if (trail is null || trail.IsDisposed)
        trail = new Trail(Main.instance.GraphicsDevice, count, new NoTip(),
            t => MathHelper.Lerp(p.WidthEnd, p.WidthStart, t),
            uv => Color.White * (1f - uv.X));

    trail.Positions = positions;
    trail.NextPosition = start + Vector2.Normalize(start - end) * 4f;

    Effect fx = ShaderLoader.GetShader("BeamTrail").Value;
    var world = Matrix.CreateTranslation(-Main.screenPosition.ToVector3());
    var view = Main.GameViewMatrix.TransformationMatrix;
    var proj = Matrix.CreateOrthographicOffCenter(0, Main.screenWidth, Main.screenHeight, 0, -1, 1);

    fx.Parameters["time"].SetValue((Main.GameUpdateCount % 10000) * p.TimeScale);
    fx.Parameters["repeats"].SetValue(p.Repeats);
    fx.Parameters["intensity"].SetValue(p.Intensity);
    fx.Parameters["noiseScale"].SetValue(p.NoiseScale);
    fx.Parameters["noiseSpeed"].SetValue(p.NoiseSpeed);
    fx.Parameters["distort"].SetValue(p.Distort);
    fx.Parameters["flicker"].SetValue(p.Flicker);
    fx.Parameters["coreWidth"].SetValue(p.CoreWidth);
    fx.Parameters["edgeWidth"].SetValue(p.EdgeWidth);
    fx.Parameters["coreColor"].SetValue(p.CoreColor);
    fx.Parameters["edgeColor"].SetValue(p.EdgeColor);
    fx.Parameters["transformMatrix"].SetValue(world * view * proj);
    fx.Parameters["sampleTexture"].SetValue(ramp);
    fx.Parameters["noiseTexture"].SetValue(noise);

    var gd = Main.graphics.GraphicsDevice;
    var oldBlend = gd.BlendState;
    gd.BlendState = p.Additive ? BlendState.Additive : BlendState.AlphaBlend;
    trail.Render(fx);
    gd.BlendState = oldBlend;
}

    Main.spriteBatch.Begin();
}

How To Implement Shaders (minimal debugging)
•
Author HLSL with explicit parameter names; keep logic simple and bounded.
•
Compile .fx into .xnb through your content pipeline; the runtime loads .xnb effects by name.
•
Load effects once, set all uniforms every frame immediately before Apply() or Render().
•
For trails, build world * view * projection and pass transformMatrix.
•
For muzzle/impact, draw a single billboard sprite with SpriteSortMode.Immediate.
•
Keep time bounded with % or fmod and keep distortion amplitudes small.
•
Use smoothstep for edges; hard steps cause shimmer and aliasing.
•
Explicitly restore SpriteBatch and BlendState after custom draws.
